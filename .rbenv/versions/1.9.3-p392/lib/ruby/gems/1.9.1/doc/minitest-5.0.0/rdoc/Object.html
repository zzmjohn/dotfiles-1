def assert_triggered expected = "blah", klass = Minitest::Assertion
  @assertion_count += 2

  e = assert_raises(klass) do
    yield
  end

  msg = e.message.sub(/(---Backtrace---).*/, '\1')
  msg.gsub!(/\(oid=[-0-9]+\)/, "(oid=N)")

  assert_equal expected, msg
end
