Gem::MockGemUi

This Gem::StreamUI subclass records input and output to StringIO for retrieval during tests.

def initialize(input = "")
  ins = StringIO.new input
  outs = StringIO.new
  errs = StringIO.new

  ins.extend TTY
  outs.extend TTY
  errs.extend TTY

  super ins, outs, errs, true

  @terminated = false
end
